import numpy as np

#import matlab data
time_domain = loadmat("../../zip_meiners/behav/dsm_l1/data/dat_time_domain_plot.mat")

#to access imported matlab data
time = time_domain['t']
voltage = time_domain['u']


plt.figure(1)
